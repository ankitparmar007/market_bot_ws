DROP VIEW IF EXISTS market.volume_1m_mv;
DROP VIEW IF EXISTS market.ticks_mv;
DROP TABLE IF EXISTS market.volume_1m;
DROP TABLE IF EXISTS market.ticks_enriched;
DROP TABLE IF EXISTS market.ticks;


CREATE TABLE market.ticks
(
    symbol LowCardinality(String),
    ts DateTime64(3, 'Asia/Kolkata'),
    ltp Float32,
    vtt UInt64
)
ENGINE = MergeTree
PARTITION BY toDate(ts)
ORDER BY (symbol, ts);


🥇 Step 1 — Create Intermediate Table (Tick With Delta)
CREATE TABLE market.ticks_enriched
(
    symbol LowCardinality(String),
    ts DateTime64(3, 'Asia/Kolkata'),
    minute DateTime('Asia/Kolkata'),
    tick_volume UInt64,
    side Enum8('buy' = 1, 'sell' = 2, 'neutral' = 3)
)
ENGINE = MergeTree
PARTITION BY toDate(ts)
ORDER BY (symbol, ts);


🥇 Step 2 — Materialized View From Raw Ticks

This calculates:

tick_volume = vtt - previous vtt

side = compare ltp

CREATE MATERIALIZED VIEW market.ticks_mv
TO market.ticks_enriched
AS
SELECT
    symbol,
    ts,
    toStartOfMinute(ts) AS minute,

    greatest(
        vtt - lagInFrame(vtt) OVER w,
        0
    ) AS tick_volume,

    if(
        ltp > lagInFrame(ltp) OVER w, 'buy',
        if(ltp < lagInFrame(ltp) OVER w, 'sell', 'neutral')
    ) AS side

FROM market.ticks
WINDOW w AS (
    PARTITION BY symbol
    ORDER BY ts
);

🥈 Step 3 — Create Final 1-Minute Volume Table
CREATE TABLE market.volume_1m
(
    symbol LowCardinality(String),
    minute DateTime('Asia/Kolkata'),
    buy UInt64,
    sell UInt64,
    total UInt64,
    delta Int64
)
ENGINE = SummingMergeTree
PARTITION BY toDate(minute)
ORDER BY (symbol, minute);


🥈 Step 4 — Second Materialized View (Minute Aggregation)
CREATE MATERIALIZED VIEW market.volume_1m_mv
TO market.volume_1m
AS
SELECT
    symbol,
    minute,

    sumIf(tick_volume, side = 'buy')  AS buy,
    sumIf(tick_volume, side = 'sell') AS sell,

    sum(tick_volume) AS total,
    sumIf(tick_volume, side = 'buy')
      - sumIf(tick_volume, side = 'sell') AS delta

FROM market.ticks_enriched
GROUP BY symbol, minute;
